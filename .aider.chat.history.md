        lucide.createIcons();
        let currentMode = 'search';
        const API_KEY_STORAGE_NAME = "SCFAI_FRONTEND_KEY";

        // ==========================================

        function toggleSidebar() {

        function openApiModal() {

            status.className = "status-msg";

        function saveApiKey() {
            const key = document.getElementById('userApiKeyInput').value.trim();
            const status = document.getElementById('saveStatus');
            
            if (key === '') {
                status.innerText = "Please enter a valid key!";
                status.className = "status-msg error";
                return;
            }
            
            localStorage.setItem(API_KEY_STORAGE_NAME, key);

        // ==========================================
        // MODE SWITCH
        // ==========================================
        function switchMode(m) {
            currentMode = m;
            document.querySelectorAll('.tab-btn').forEach(b => {
                b.classList.toggle('active', b.innerText.toLowerCase().includes(m === 'search' ? 'clip' : 'script'));
            });
            document.getElementById('searchView').style.display = m === 'search' ? 'block' : 'none';
            document.getElementById('timelineView').style.display = m === 'timeline' ? 'block' : 'none';
            document.getElementById('runErrorStatus').className = "status-msg";

        function openResults() { document.getElementById('resultsOverlay').style.display = 'flex'; }

        // ==========================================
        // AI LOGIC
        // ==========================================
        const steps = ["Groq Brainstorming...", "FAISS Mapping...", "CLIP Vision Checking...", "Finalizing HD Clips..."];

        async function startAI() {
            const userKey = localStorage.getItem(API_KEY_STORAGE_NAME);
            const runStatus = document.getElementById('runErrorStatus');
            
            if (!userKey) {
                runStatus.innerText = "Please save your API Key in settings first!";
                runStatus.className = "status-msg error";
                setTimeout(() => openApiModal(), 500);
                return;
            }

            const loader = document.getElementById('loader');
            const loaderText = document.getElementById('loaderText');
            loader.style.display = 'flex';
            
            let sIdx = 0;
            const timer = setInterval(() => {
                loaderText.innerText = steps[sIdx % steps.length];
                sIdx++;
            }, 2000);

            const payload = currentMode === 'search'
                ? { query: document.getElementById('queryInput').value, orientation: document.getElementById('orientation').value, quality: document.getElementById('quality').value }
                : { script: document.getElementById('scriptInput').value, orientation: document.getElementById('orientation').value, quality: document.getElementById('quality').value };

            const endpoint = currentMode === 'search' ? '/api/search' : '/api/timeline';

            try {
                const res = await fetch(endpoint, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'X-API-Key': userKey },
                    body: JSON.stringify(payload)
                });
                const data = await res.json();
                
                if (res.ok && data.success) {
                    pollStatus(data.task_id, timer, userKey);
                } else {
                    clearInterval(timer);
                    loader.style.display = 'none';
                    const errMsg = data.detail || data.error || "Request failed";
                    runStatus.innerText = (res.status === 401 || res.status === 403) ? "Unauthorized: Invalid API Key!" : errMsg;
                    runStatus.className = "status-msg error";
                }
            } catch (err) {
                clearInterval(timer);
                loader.style.display = 'none';
                runStatus.innerText = "Connection Failed. Backend might be sleeping.";
                runStatus.className = "status-msg error";
            }
        }

        async function pollStatus(tid, timer, userKey) {
            const runStatus = document.getElementById('runErrorStatus');
            const poll = setInterval(async () => {
                try {
                    const res = await fetch(`/api/status/${tid}`, { headers: { 'X-API-Key': userKey } });
                    const data = await res.json();
                    
                    if (data.task && data.task.status === 'completed') {
                        clearInterval(poll); clearInterval(timer);
                        document.getElementById('loader').style.display = 'none';
                        renderResults(data.task.data);
                    } else if (data.task && data.task.status === 'error') {
                        clearInterval(poll); clearInterval(timer);
                        document.getElementById('loader').style.display = 'none';
                        runStatus.innerText = data.task.error || "Task failed";
                        runStatus.className = "status-msg error";
                    } else if (!res.ok) {
                        clearInterval(poll); clearInterval(timer);
                        document.getElementById('loader').style.display = 'none';
                        runStatus.innerText = data.detail || "Polling failed";
                        runStatus.className = "status-msg error";
                    }
                } catch(err) { console.log("Polling..."); }
            }, 3000);
        }

            if (currentMode === 'search') {
                data.forEach(clip => {
                    container.innerHTML += `
                        <div class="video-card">
                            <div class="video-thumb" style="${aspectStyle}">
                                <video src="${clip.download_url}" poster="${clip.thumbnail}" muted loop onmouseover="this.play()" onmouseout="this.pause()" playsinline></video>
                                <span class="badge">${clip.quality_label}</span>
                            </div>
                            <div class="card-info">
                                <div class="ai-meta">
                                    <span><span class="tag-chip">CLIP Vision: ${clip.visual_score}%</span></span>
                                    <span>🔍 ${clip.query_used}</span>

                const data = await res.json();
                
                if (res.ok && data.success) {
                    pollStatus(data.task_id, timer, userKey);
                } else {
                    clearInterval(timer);
                    loader.style.display = 'none';
                    const errMsg = data.detail || data.error || "Request failed";
                    runStatus.innerText = (res.status === 401 || res.status === 403) ? "Unauthorized: Invalid API Key!" : errMsg;
                    runStatus.className = "status-msg error";
                }
            } catch (err) {
                clearInterval(timer);
                loader.style.display = 'none';
                runStatus.innerText = "Connection Failed. Backend might be sleeping.";
                runStatus.className = "status-msg error";
            }
        }

        async function pollStatus(tid, timer, userKey) {
            const runStatus = document.getElementById('runErrorStatus');
            const poll = setInterval(async () => {
                try {
                    const res = await fetch(`/api/status/${tid}`, { headers: { 'X-API-Key': userKey } });
                    const data = await res.json();
                    
                    if (data.task && data.task.status === 'completed') {
                        clearInterval(poll); clearInterval(timer);
                        document.getElementById('loader').style.display = 'none';
                        renderResults(data.task.data);
                    } else if (data.task && data.task.status === 'error') {
                        clearInterval(poll); clearInterval(timer);
                        document.getElementById('loader').style.display = 'none';
                        runStatus.innerText = data.task.error || "Task failed";
                        runStatus.className = "status-msg error";
                    } else if (!res.ok) {
                        clearInterval(poll); clearInterval(timer);
                        document.getElementById('loader').style.display = 'none';
                        runStatus.innerText = data.detail || "Polling failed";
                        runStatus.className = "status-msg error";
                    }
                } catch(err) { console.log("Polling..."); }
            }, 3000);
        }
